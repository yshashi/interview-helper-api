# Optimizing CI/CD Performance

## 1. Caching Dependencies

```yaml
# GitHub Actions caching
jobs:
  build:
    steps:
      - uses: actions/cache@v3
        with:
          path: |
            ~/.npm
            ~/.m2
            ~/.nuget
          key: ${{ runner.os }}-deps-${{ hashFiles('**/package-lock.json', '**/pom.xml', '**/packages.lock.json') }}
          restore-keys: |
            ${{ runner.os }}-deps-
      
      - run: npm ci  # Uses cache if available
```

## 2. Parallel Execution

```yaml
# Run jobs in parallel
jobs:
  unit-tests:
    runs-on: ubuntu-latest
    steps:
      - run: npm run test:unit
  
  integration-tests:
    runs-on: ubuntu-latest
    steps:
      - run: npm run test:integration
  
  e2e-tests:
    runs-on: ubuntu-latest
    steps:
      - run: npm run test:e2e
  
  lint:
    runs-on: ubuntu-latest
    steps:
      - run: npm run lint
```

## 3. Matrix Builds

```yaml
# Test multiple versions in parallel
jobs:
  test:
    strategy:
      matrix:
        node: [16, 18, 20]
        os: [ubuntu-latest, windows-latest]
    runs-on: ${{ matrix.os }}
    steps:
      - uses: actions/setup-node@v3
        with:
          node-version: ${{ matrix.node }}
      - run: npm test
```

## 4. Incremental Builds

```yaml
# Only build changed components
jobs:
  detect-changes:
    outputs:
      frontend: ${{ steps.changes.outputs.frontend }}
      api: ${{ steps.changes.outputs.api }}
    steps:
      - uses: dorny/paths-filter@v2
        id: changes
        with:
          filters: |
            frontend:
              - 'frontend/**'
            api:
              - 'api/**'
  
  build-frontend:
    needs: detect-changes
    if: needs.detect-changes.outputs.frontend == 'true'
    steps:
      - run: cd frontend && npm run build
  
  build-api:
    needs: detect-changes
    if: needs.detect-changes.outputs.api == 'true'
    steps:
      - run: cd api && dotnet build
```

## 5. Docker Layer Caching

```dockerfile
# Optimize Docker layers
FROM node:18-alpine AS deps
WORKDIR /app
COPY package*.json ./
RUN npm ci --only=production

FROM node:18-alpine AS build
WORKDIR /app
COPY package*.json ./
RUN npm ci
COPY . .
RUN npm run build

FROM node:18-alpine AS production
WORKDIR /app
COPY --from=deps /app/node_modules ./node_modules
COPY --from=build /app/dist ./dist
CMD ["node", "dist/index.js"]
```

## 6. Selective Testing

```javascript
// Run only affected tests
const { execSync } = require('child_process');

const changedFiles = execSync('git diff --name-only HEAD~1')
  .toString()
  .split('\n')
  .filter(Boolean);

const testFiles = changedFiles
  .filter(file => file.endsWith('.spec.ts'))
  .join(' ');

if (testFiles) {
  execSync(`jest ${testFiles}`, { stdio: 'inherit' });
} else {
  console.log('No test files changed, skipping tests');
}
```

## 7. Artifact Optimization

```yaml
# Only upload necessary artifacts
jobs:
  build:
    steps:
      - run: npm run build
      
      - uses: actions/upload-artifact@v3
        with:
          name: dist
          path: dist/
          retention-days: 7  # Auto-delete after 7 days
          if-no-files-found: error
```

## 9. Fail Fast

```yaml
jobs:
  lint:
    steps:
      - run: npm run lint
  
  test:
    needs: lint  # Don't run if lint fails
    steps:
      - run: npm test
  
  build:
    needs: test  # Don't build if tests fail
    steps:
      - run: npm run build
```

## 14. Reduce Image Size

```dockerfile
# Multi-stage build
FROM node:18 AS build
WORKDIR /app
COPY . .
RUN npm ci && npm run build

FROM node:18-alpine  # Smaller base image
WORKDIR /app
COPY --from=build /app/dist ./dist
RUN npm ci --only=production
CMD ["node", "dist/index.js"]

# Result: 150MB vs 1.2GB
```

## 15. Pipeline Monitoring

```javascript
// Track pipeline metrics
const metrics = {
  startTime: Date.now(),
  stages: {}
};

async function trackStage(name, fn) {
  const start = Date.now();
  await fn();
  metrics.stages[name] = Date.now() - start;
}

await trackStage('build', async () => {
  await exec('npm run build');
});

await trackStage('test', async () => {
  await exec('npm test');
});

console.log('Pipeline metrics:', metrics);
```
