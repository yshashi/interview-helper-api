## 1. Database Optimization

```javascript
// ❌ Bad - N+1 query problem
app.get('/api/users', async (req, res) => {
  const users = await User.find();
  
  for (const user of users) {
    user.orders = await Order.find({ userId: user.id });
  }
  
  res.json(users);
});

// ✅ Good - Use joins/populate
app.get('/api/users', async (req, res) => {
  const users = await User.find().populate('orders');
  res.json(users);
});

// ✅ Good - Aggregation
app.get('/api/users', async (req, res) => {
  const users = await User.aggregate([
    {
      $lookup: {
        from: 'orders',
        localField: '_id',
        foreignField: 'userId',
        as: 'orders'
      }
    }
  ]);
  
  res.json(users);
});
```

## 2. Indexing

```javascript
// Create indexes
const userSchema = new mongoose.Schema({
  email: { type: String, index: true, unique: true },
  name: { type: String, index: true },
  createdAt: { type: Date, index: true }
});

// Compound index
userSchema.index({ role: 1, status: 1 });

// Text index for search
userSchema.index({ name: 'text', email: 'text' });

// Usage
const users = await User.find({ role: 'admin', status: 'active' });
```

```csharp
// .NET EF Core indexes
protected override void OnModelCreating(ModelBuilder modelBuilder)
```

## 3. Caching

```javascript
const redis = require('redis');
const client = redis.createClient();

// Cache-aside pattern
app.get('/api/users/:id', async (req, res) => {
  const cacheKey = `user:${req.params.id}`;
  
  // Try cache
  const cached = await client.get(cacheKey);
  if (cached) {
    return res.json(JSON.parse(cached));
  }
  
  // Load from database
  const user = await User.findById(req.params.id);
  
  // Store in cache (5 minutes)
  await client.setex(cacheKey, 300, JSON.stringify(user));
  
  res.json(user);
});

// HTTP caching
app.get('/api/users/:id', async (req, res) => {
  const user = await User.findById(req.params.id);
  
  res.set('Cache-Control', 'public, max-age=300');
  res.set('ETag', generateETag(user));
  
  res.json(user);
});
```

## 4. Pagination

```javascript
// ❌ Bad - Load all data
app.get('/api/users', async (req, res) => {
  const users = await User.find(); // Could be millions
  res.json(users);
});

// ✅ Good - Paginate
app.get('/api/users', async (req, res) => {
  const page = parseInt(req.query.page) || 1;
  const limit = parseInt(req.query.limit) || 10;
  const skip = (page - 1) * limit;
  
  const users = await User.find()
    .skip(skip)
    .limit(limit);
  
  res.json({ data: users });
});

// ✅ Better - Cursor-based pagination
app.get('/api/users', async (req, res) => {
  const limit = parseInt(req.query.limit) || 10;
  const cursor = req.query.cursor;
  
  const query = cursor ? { _id: { $gt: cursor } } : {};
  const users = await User.find(query).limit(limit);
  
  res.json({
    data: users,
    nextCursor: users.length > 0 ? users[users.length - 1]._id : null
  });
});
```

## 5. Field Selection

```javascript
// ❌ Bad - Return all fields
app.get('/api/users', async (req, res) => {
  const users = await User.find();
  res.json(users);
});

// ✅ Good - Select only needed fields
app.get('/api/users', async (req, res) => {
  const users = await User.find().select('name email role');
  res.json(users);
});

// ✅ Good - Client-specified fields
app.get('/api/users', async (req, res) => {
  const fields = req.query.fields?.split(',').join(' ') || 'name email';
  const users = await User.find().select(fields);
  res.json(users);
});
```

## 8. Async Processing

## 11. Database Query Optimization

```javascript
// ❌ Bad - Multiple queries
const user = await User.findById(userId);
const orders = await Order.find({ userId });
const payments = await Payment.find({ userId });

// ✅ Good - Parallel queries
const [user, orders, payments] = await Promise.all([
  User.findById(userId),
  Order.find({ userId }),
  Payment.find({ userId })
]);

// ✅ Good - Single aggregation
const result = await User.aggregate([
  { $match: { _id: userId } },
  {
    $lookup: {
      from: 'orders',
      localField: '_id',
      foreignField: 'userId',
      as: 'orders'
    }
  },
  {
    $lookup: {
      from: 'payments',
      localField: '_id',
      foreignField: 'userId',
      as: 'payments'
    }
  }
]);
```

## 12. Response Streaming

```javascript
// Stream large datasets
app.get('/api/users/export', async (req, res) => {
  res.setHeader('Content-Type', 'application/json');
  res.write('[');
  
  const cursor = User.find().cursor();
  let first = true;
  
  for await (const user of cursor) {
    if (!first) res.write(',');
    res.write(JSON.stringify(user));
    first = false;
  }
  
  res.write(']');
  res.end();
});
```

## 14. Monitoring and Profiling

```javascript
// Profile slow queries
mongoose.set('debug', (collectionName, method, query, doc) => {
  console.log(`${collectionName}.${method}`, JSON.stringify(query));
});

// Track response times
app.use((req, res, next) => {
  const start = Date.now();
  
  res.on('finish', () => {
    const duration = Date.now() - start;
    if (duration > 1000) {
      console.warn(`Slow request: ${req.method} ${req.path} - ${duration}ms`);
    }
  });
  
  next();
});
```
