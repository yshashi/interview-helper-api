## Service Communication

### 1. Synchronous (REST)

```javascript
// User Service
app.get('/api/users/:id', async (req, res) => {
  const user = await User.findById(req.params.id);
  res.json(user);
});

// Order Service - calls User Service
app.get('/api/orders/:id', async (req, res) => {
  const order = await Order.findById(req.params.id);
  
  // Fetch user data from User Service
  const userResponse = await fetch(`http://user-service:3001/api/users/${order.userId}`);
  const user = await userResponse.json();
  
  res.json({
    ...order,
    user
  });
});
```

### 2. Asynchronous (Message Queue)

```javascript
// Order Service - publishes event
const amqp = require('amqplib');

async function createOrder(orderData) {
  const order = await Order.create(orderData);
  
  // Publish event
  const connection = await amqp.connect(process.env.RABBITMQ_URL);
  const channel = await connection.createChannel();
  
  await channel.assertExchange('orders', 'topic', { durable: true });
  channel.publish('orders', 'order.created', Buffer.from(JSON.stringify(order)));
  
  return order;
}

// Payment Service - subscribes to events
async function subscribeToOrders() {
  const connection = await amqp.connect(process.env.RABBITMQ_URL);
  const channel = await connection.createChannel();
  
  await channel.assertExchange('orders', 'topic', { durable: true });
  const queue = await channel.assertQueue('payment-orders', { durable: true });
  
  await channel.bindQueue(queue.queue, 'orders', 'order.created');
  
  channel.consume(queue.queue, async (msg) => {
    const order = JSON.parse(msg.content.toString());
    await processPayment(order);
    channel.ack(msg);
  });
}
```

## API Gateway Pattern

```javascript
const express = require('express');
const { createProxyMiddleware } = require('http-proxy-middleware');

const app = express();

// Authentication middleware
const authenticate = async (req, res, next) => {
  const token = req.headers.authorization?.replace('Bearer ', '');
  if (!token) {
    return res.status(401).json({ error: 'Unauthorized' });
  }
  
  try {
    const decoded = jwt.verify(token, process.env.JWT_SECRET);
    req.user = decoded;
    next();
  } catch (error) {
    res.status(401).json({ error: 'Invalid token' });
  }
};

// Rate limiting
const limiter = rateLimit({
  windowMs: 15 * 60 * 1000,
  max: 100
});

// Route to services
app.use('/api/users', authenticate, limiter, createProxyMiddleware({
  target: 'http://user-service:3001',
  changeOrigin: true,
  pathRewrite: { '^/api/users': '' }
}));

app.use('/api/orders', authenticate, limiter, createProxyMiddleware({
  target: 'http://order-service:3002',
  changeOrigin: true,
  pathRewrite: { '^/api/orders': '' }
}));

app.use('/api/payments', authenticate, limiter, createProxyMiddleware({
  target: 'http://payment-service:3003',
  changeOrigin: true,
  pathRewrite: { '^/api/payments': '' }
}));

app.listen(3000);
```

## Service Discovery

```javascript
const Consul = require('consul');
const consul = new Consul();

// Register service
async function registerService() {
  await consul.agent.service.register({
    id: 'user-service-1',
    name: 'user-service',
    address: 'localhost',
    port: 3001,
    check: {
      http: 'http://localhost:3001/health',
      interval: '10s'
    }
  });
}

// Discover service
async function getServiceUrl(serviceName) {
  const services = await consul.health.service({
    service: serviceName,
    passing: true
  });
  
  if (services.length === 0) {
    throw new Error(`No healthy instances of ${serviceName}`);
  }
  
  const service = services[Math.floor(Math.random() * services.length)];
  return `http://${service.Service.Address}:${service.Service.Port}`;
}

// Use in API Gateway
app.use('/api/users', async (req, res, next) => {
  try {
    const target = await getServiceUrl('user-service');
    createProxyMiddleware({ target, changeOrigin: true })(req, res, next);
  } catch (error) {
    res.status(503).json({ error: 'Service unavailable' });
  }
});
```

## Circuit Breaker

```javascript
const CircuitBreaker = require('opossum');

const options = {
  timeout: 3000,
  errorThresholdPercentage: 50,
  resetTimeout: 30000
};

const callUserService = async (userId) => {
  const response = await fetch(`http://user-service:3001/api/users/${userId}`);
  return response.json();
};

const breaker = new CircuitBreaker(callUserService, options);

breaker.on('open', () => {
  console.log('Circuit breaker opened for user-service');
});

breaker.on('halfOpen', () => {
  console.log('Circuit breaker half-open for user-service');
});

app.get('/api/orders/:id', async (req, res) => {
  const order = await Order.findById(req.params.id);
  
  try {
    const user = await breaker.fire(order.userId);
    res.json({ ...order, user });
  } catch (error) {
    // Fallback: return order without user data
    res.json(order);
  }
});
```

## Saga Pattern

```javascript
// Order Service - Orchestrator
class OrderSaga {
  async createOrder(orderData) {
    const order = await Order.create({ ...orderData, status: 'pending' });
    
    try {
      // Step 1: Reserve inventory
      await this.reserveInventory(order);
      
      // Step 2: Process payment
      await this.processPayment(order);
      
      // Step 3: Confirm order
      order.status = 'confirmed';
      await order.save();
      
      return order;
    } catch (error) {
      // Compensating transactions
      await this.compensate(order);
      throw error;
    }
  }
  
  async reserveInventory(order) {
    const response = await fetch('http://inventory-service:3004/api/reserve', {
      method: 'POST',
      body: JSON.stringify({ orderId: order.id, items: order.items })
    });
    
    if (!response.ok) {
      throw new Error('Inventory reservation failed');
    }
  }
  
  async processPayment(order) {
    const response = await fetch('http://payment-service:3003/api/charge', {
      method: 'POST',
      body: JSON.stringify({ orderId: order.id, amount: order.total })
    });
    
    if (!response.ok) {
      throw new Error('Payment failed');
    }
  }
  
  async compensate(order) {
    // Rollback inventory
    await fetch('http://inventory-service:3004/api/release', {
      method: 'POST',
      body: JSON.stringify({ orderId: order.id })
    });
    
    // Refund payment if charged
    await fetch('http://payment-service:3003/api/refund', {
      method: 'POST',
      body: JSON.stringify({ orderId: order.id })
    });
    
    order.status = 'cancelled';
    await order.save();
  }
}
```

## Backend for Frontend (BFF)

```javascript
// Mobile BFF
app.get('/mobile/api/dashboard', async (req, res) => {
  const [user, orders, notifications] = await Promise.all([
    fetch(`http://user-service:3001/api/users/${req.user.id}`),
    fetch(`http://order-service:3002/api/orders?userId=${req.user.id}&limit=5`),
    fetch(`http://notification-service:3005/api/notifications?userId=${req.user.id}&limit=3`)
  ]);
  
  res.json({
    user: await user.json(),
    recentOrders: await orders.json(),
    notifications: await notifications.json()
  });
});

// Web BFF
app.get('/web/api/dashboard', async (req, res) => {
```

## Distributed Tracing

```javascript
const opentelemetry = require('@opentelemetry/api');
const { NodeTracerProvider } = require('@opentelemetry/sdk-trace-node');
const { JaegerExporter } = require('@opentelemetry/exporter-jaeger');

const provider = new NodeTracerProvider();
const exporter = new JaegerExporter({
  serviceName: 'order-service'
});

provider.addSpanProcessor(new opentelemetry.SimpleSpanProcessor(exporter));
provider.register();

const tracer = opentelemetry.trace.getTracer('order-service');

app.post('/api/orders', async (req, res) => {
  const span = tracer.startSpan('create-order');
  
  try {
    const order = await Order.create(req.body);
    
    // Call user service with trace context
    const userSpan = tracer.startSpan('fetch-user', { parent: span });
    const user = await fetch(`http://user-service:3001/api/users/${order.userId}`, {
      headers: {
        'traceparent': userSpan.spanContext().traceId
      }
    });
    userSpan.end();
    
    res.status(201).json(order);
  } finally {
    span.end();
  }
});
```

## Interview Tips

- **Explain microservices**: Independent, scalable services
- **Show communication**: Sync (REST) vs async (messaging)
- **Demonstrate gateway**: Single entry point
- **Discuss patterns**: Circuit breaker, saga, BFF
- **Mention discovery**: Service registry
- **Show orchestration**: Docker Compose, Kubernetes

## Summary

Microservices architecture splits monolith into independent services communicating via REST APIs or message queues. Use API Gateway as single entry point for authentication, rate limiting, and routing. Implement service discovery with Consul or Kubernetes. Use circuit breakers for resilience. Apply saga pattern for distributed transactions. Create BFF for different clients. Enable distributed tracing. Deploy with Docker and Kubernetes. Essential for scalable, maintainable systems.
