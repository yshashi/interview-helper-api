## MongoDB Production Setup

### Replica Set Configuration

```yaml
# docker-compose.yml
version: '3.8'

services:
  mongo-primary:
    image: mongo:7
    command: mongod --replSet rs0 --bind_ip_all
    ports:
      - "27017:27017"
    volumes:
      - mongo-primary-data:/data/db
    environment:
      MONGO_INITDB_ROOT_USERNAME: admin
      MONGO_INITDB_ROOT_PASSWORD: ${MONGO_PASSWORD}
  
  mongo-secondary1:
    image: mongo:7
    command: mongod --replSet rs0 --bind_ip_all
    ports:
      - "27018:27017"
    volumes:
      - mongo-secondary1-data:/data/db
    depends_on:
      - mongo-primary
  
  mongo-secondary2:
    image: mongo:7
    command: mongod --replSet rs0 --bind_ip_all
    ports:
      - "27019:27017"
    volumes:
      - mongo-secondary2-data:/data/db
    depends_on:
      - mongo-primary

volumes:
  mongo-primary-data:
  mongo-secondary1-data:
  mongo-secondary2-data:
```

## Kubernetes Deployment

### MongoDB StatefulSet

```yaml
# mongodb-statefulset.yaml
apiVersion: apps/v1
kind: StatefulSet
metadata:
  name: mongodb
spec:
  serviceName: mongodb
  replicas: 3
  selector:
    matchLabels:
      app: mongodb
  template:
    metadata:
      labels:
        app: mongodb
    spec:
      containers:
      - name: mongodb
        image: mongo:7
        command:
          - mongod
          - "--replSet"
          - rs0
          - "--bind_ip_all"
        ports:
        - containerPort: 27017
        volumeMounts:
        - name: mongodb-data
          mountPath: /data/db
        env:
        - name: MONGO_INITDB_ROOT_USERNAME
          valueFrom:
            secretKeyRef:
              name: mongodb-secret
              key: username
        - name: MONGO_INITDB_ROOT_PASSWORD
          valueFrom:
            secretKeyRef:
              name: mongodb-secret
              key: password
  volumeClaimTemplates:
  - metadata:
      name: mongodb-data
    spec:
      accessModes: [ "ReadWriteOnce" ]
      resources:
        requests:
          storage: 100Gi
```

### Connection Management

```javascript
const { MongoClient } = require('mongodb');

class DatabaseConnection {
  constructor() {
    this.client = null;
    this.db = null;
  }
  
  async connect() {
    if (this.client) {
      return this.db;
    }
    
    const config = require('./config');
    
    this.client = new MongoClient(config.mongodb.uri, config.mongodb.options);
    
    await this.client.connect();
    
    this.db = this.client.db();
    
    // Handle connection events
    this.client.on('serverHeartbeatFailed', (event) => {
      console.error('Heartbeat failed:', event);
    });
    
    this.client.on('topologyDescriptionChanged', (event) => {
      console.log('Topology changed:', event.newDescription.type);
    });
    
    return this.db;
  }
  
  async close() {
    if (this.client) {
      await this.client.close();
      this.client = null;
      this.db = null;
    }
  }
}

// Singleton instance
const dbConnection = new DatabaseConnection();

// Graceful shutdown
process.on('SIGINT', async () => {
  await dbConnection.close();
  process.exit(0);
});

module.exports = dbConnection;
```

## Health Checks

### Kubernetes Probes

```yaml
# deployment.yaml
apiVersion: apps/v1
kind: Deployment
metadata:
  name: myapp
spec:
  replicas: 3
  template:
    spec:
      containers:
      - name: myapp
        image: myapp:latest
        ports:
        - containerPort: 3000
        livenessProbe:
          httpGet:
            path: /health/live
            port: 3000
          initialDelaySeconds: 30
          periodSeconds: 10
        readinessProbe:
          httpGet:
            path: /health/ready
            port: 3000
          initialDelaySeconds: 5
          periodSeconds: 5
```

### Health Check Endpoints

```javascript
// health.js
const express = require('express');
const router = express.Router();

router.get('/health/live', (req, res) => {
  res.json({ status: 'ok' });
});

router.get('/health/ready', async (req, res) => {
  try {
    // Check MongoDB
    await db.admin().ping();
    
    // Check Redis
    await redisClient.ping();
    
    res.json({
      status: 'ready',
      mongodb: 'connected',
      redis: 'connected'
    });
  } catch (error) {
    res.status(503).json({
      status: 'not ready',
      error: error.message
    });
  }
});

module.exports = router;
```

## Zero-Downtime Deployment

### Blue-Green Deployment

```javascript
// Deploy new version (green)
// 1. Deploy green environment
// 2. Run smoke tests
// 3. Switch traffic to green
// 4. Keep blue for rollback

const deploymentStrategy = {
  blueGreen: {
    steps: [
      'Deploy green environment',
      'Warm up connections',
      'Run health checks',
      'Switch load balancer to green',
      'Monitor for errors',
      'Decommission blue (after verification)'
    ]
  }
};
```

## Database Migration in Production

```javascript
class ProductionMigration {
  async migrate() {
    console.log('Starting production migration...');
    
    // 1. Create backup
    await this.createBackup();
    
    // 2. Run migration in transaction
    const session = client.startSession();
    
    try {
      await session.withTransaction(async () => {
        await this.runMigration(session);
      });
      
      console.log('Migration completed successfully');
    } catch (error) {
      console.error('Migration failed:', error);
      await this.rollback();
      throw error;
    } finally {
      await session.endSession();
    }
  }
  
  async runMigration(session) {
    // Add new field with default value
    await db.collection('users').updateMany(
      { version: { $exists: false } },
      {
        $set: {
          status: 'active',
          version: 2
        }
      },
      { session }
    );
    
    // Create new index
    await db.collection('users').createIndex(
      { status: 1 },
      { session }
    );
  }
}
```

## Monitoring in Production

```javascript
// Prometheus metrics
const client = require('prom-client');
const register = new client.Registry();

// Database metrics
const dbConnectionGauge = new client.Gauge({
  name: 'mongodb_connections_current',
  help: 'Current MongoDB connections',
  registers: [register]
});

const queryDurationHistogram = new client.Histogram({
  name: 'mongodb_query_duration_seconds',
  help: 'MongoDB query duration',
  labelNames: ['operation', 'collection'],
  registers: [register]
});

// Collect metrics
setInterval(async () => {
  const status = await db.adminCommand({ serverStatus: 1 });
  dbConnectionGauge.set(status.connections.current);
}, 10000);

// Expose metrics
app.get('/metrics', async (req, res) => {
  res.set('Content-Type', register.contentType);
  res.end(await register.metrics());
});
```

```javascript
const { SecretsManagerClient, GetSecretValueCommand } = require('@aws-sdk/client-secrets-manager');

async function getMongoDBCredentials() {
  const client = new SecretsManagerClient({ region: 'us-east-1' });
  
  const command = new GetSecretValueCommand({
```

## .NET Production Deployment

```csharp
using MongoDB.Driver;
using Microsoft.Extensions.Options;

public class MongoDbService
{
    private readonly IMongoClient _client;
    private readonly IMongoDatabase _database;
    
    public MongoDbService(IOptions<MongoDbSettings> settings)
    {
        var mongoSettings = MongoClientSettings.FromConnectionString(
            settings.Value.ConnectionString
        );
        
        // Production settings
        mongoSettings.MaxConnectionPoolSize = 100;
        mongoSettings.MinConnectionPoolSize = 10;
        mongoSettings.ServerSelectionTimeout = TimeSpan.FromSeconds(5);
        mongoSettings.ConnectTimeout = TimeSpan.FromSeconds(10);
        mongoSettings.SocketTimeout = TimeSpan.FromSeconds(30);
        
        // TLS
        mongoSettings.UseTls = true;
        
        // Write concern
        mongoSettings.WriteConcern = WriteConcern.WMajority;
        
        // Read preference
        mongoSettings.ReadPreference = ReadPreference.SecondaryPreferred;
        
        _client = new MongoClient(mongoSettings);
        _database = _client.GetDatabase(settings.Value.DatabaseName);
    }
}
```

## Interview Tips

- **Explain replica sets**: High availability setup
- **Show Kubernetes**: StatefulSets, services
- **Demonstrate health checks**: Liveness, readiness probes
- **Discuss deployment strategies**: Blue-green, rolling
- **Mention monitoring**: Prometheus, metrics
- **Show configuration**: Environment-based settings

## Summary

Deploy NoSQL databases in production with replica sets for high availability. Use Kubernetes StatefulSets for orchestration. Configure connection pooling, TLS, and authentication. Implement health checks for liveness and readiness. Use blue-green or rolling deployment strategies for zero downtime. Monitor with Prometheus metrics. Manage secrets securely with AWS Secrets Manager or Kubernetes Secrets. Automate backups and test disaster recovery. Document runbooks for operations. Essential for reliable production NoSQL deployments.
